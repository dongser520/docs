    // 解码f参数
    decode_f(f){
       let _fields = [];
       //解码参数
       const base64Param = f
       .replace(/-/g, '+') // 将-换回+
       .replace(/_/g, '/'); // 将_换回/

       // 补足Base64末尾的等号
       let padded = base64Param;
       while (padded.length % 4 !== 0) {
           padded += '=';
       }
       try {
           // 解码Base64得到URI编码的字符串
           const uriEncoded = Buffer.from(padded, 'base64').toString('utf8');
           // URI解码获取原始JSON字符串
           const jsonStr = decodeURIComponent(uriEncoded);
           // 解析JSON
           _fields = JSON.parse(jsonStr);
           return _fields;
       } catch (error) {
           return ctx.apiFail('无效的参数格式');
       }
    }
