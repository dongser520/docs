### 1. 聊天页设置相关信息获取

```js

// 聊天页设置相关信息获取
getChatPageSet(to_id, chatType){
    let xiaoxiList = this.getXiaoXiList();
    // 找到当前聊天
    let index = xiaoxiList.findIndex(v => v.id == to_id && v.chatType == chatType);
    if(index != -1){
        // 找到了
        return xiaoxiList[index];
    }
    return null;
}
```
